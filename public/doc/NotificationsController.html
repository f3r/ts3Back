def index
  @notifications = []
  Notification.all.each{|foo|
    @notifications << ActiveSupport::JSON::decode(foo)
    }
  if @notifications
    return_message(200, :ok, {:notifications => @notifications})
  else
    return_message(200, :ok, {:err => {:notifications => [115]}})
  end
end

def unread
  @notifications = []
  Notification.unread.each{|foo|
    @notifications << ActiveSupport::JSON::decode(foo)
    }
  if @notifications
    return_message(200, :ok, {:notifications => @notifications})
  else
    return_message(200, :ok, {:err => {:notifications => [115]}})
  end
end
